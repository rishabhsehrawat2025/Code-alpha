#include<stdio.h>
void addition(int matrix1[][10],int matrix2[][10],int row,int column){
	int result[row][column];
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++)	{
			result[i][j]=matrix1[i][j]+matrix2[i][j];
		}
	}
	printf("Addition of matrix 1 and matrix 2 : \n");
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++){
			printf("%d ",result[i][j]);
		}
		printf("\n");
	}
}
void multiplication(int matrix1[][10],int matrix2[][10],int row,int column){
	int result[row][column];
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++){
			result[i][j]=0;
		}
	}
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++){
			for(int k=0;k<column;k++){
				result[i][j]+=matrix1[i][k]*matrix2[k][j];
			}
		}
	}
	printf("Multiplication of matrix 1 and matrix 2 : \n");
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++){
			printf("%d ",result[i][j]);
		}
		printf("\n");
	}
}
void  transpose(int matrix1[][10],int matrix2[][10],int row,int column){
	printf("Transpose of matrix 1\n");
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++){
			printf("%d ",matrix1[j][i]);
		}
	printf("\n");
	}
	printf("Transpose of matrix 2\n");
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++){
			printf("%d ",matrix2[j][i]);
		}
	printf("\n");
	}
}
int main(){
	int row,column;
	printf("Enter the value of row and column: ");
	scanf("%d %d",&row,&column);
	int matrix1[10][10],matrix2[10][10];
	for(int i=0;i<row;i++){
		for(int j=0;j<column;j++){
			scanf("%d ",&matrix1[i][j]);
		}
	}
	for(int i=0;i<row;i++){
                for(int j=0;j<column;j++){
                        scanf("%d ",&matrix2[i][j]);
                }
        }
	printf("Enter your choice:\n 1 for Additon\n 2 for multiplication\n 3 for Transpose\n");
	int a;
	scanf("%d",&a);
	switch(a){
		case 1:
			addition(matrix1,matrix2,row,column);
			break;
		case 2:
			multiplication(matrix1,matrix2,row,column);
			break;
		case 3:
			transpose(matrix1,matrix2,row,column);
			break;
		default:
			printf("Alert! Invalid Operation Demanded\n");
			break;
	}
	return 0;
}
